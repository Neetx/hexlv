struct example {
	unsigned char config;	/* bit 0 - 7*/
	unsigned short address; /* bit 8 - 23 */
	unsigned char index;	/* bit 24 - 31 */
} __attribute__((packed));	PACKED è un attributo per far conservare al compilatore la grandezza precisa della struct, a livello hardware è fondamentale (non è C standard, potrebbe non funzionare sempre)
